from utils.model_generation import ModelGenerator

gen = ModelGenerator()

# Activities
customer_searches_for_ticket = gen.activity('Customer searches for ticket')
select_route = gen.activity('Select route')
select_date_and_time = gen.activity('Select date and time')
provide_personal_information = gen.activity('Provide personal information')
provide_payment_details = gen.activity('Provide payment details')
generate_ticket = gen.activity('Generate ticket')
send_ticket_via_email = gen.activity('Send ticket via email')
send_ticket_via_sms = gen.activity('Send ticket via SMS')
update_seat_inventory = gen.activity('Update seat inventory')
send_reminder = gen.activity('Send reminder')
send_instructions = gen.activity('Send instructions')
customer_completes_journey = gen.activity('Customer completes journey')
post_travel_feedback_or_services = gen.activity('Post-travel feedback or services')


# Subprocesses
ticket_delivery = gen.xor(send_ticket_via_email, send_ticket_via_sms)
optional_reminders = gen.xor(gen.partial_order(dependencies=[(send_reminder, send_instructions), (send_instructions, customer_completes_journey.copy())]), None)

# Main process
initial_selection = gen.partial_order(dependencies=[(customer_searches_for_ticket, select_route), (select_route, select_date_and_time), (select_date_and_time, provide_personal_information), (provide_personal_information, provide_payment_details)])
provide_payment_details_copy = provide_payment_details.copy()
booking_and_updates = gen.partial_order(dependencies=[(provide_payment_details_copy, generate_ticket), (generate_ticket, ticket_delivery), (generate_ticket, update_seat_inventory)])
customer_completes_journey_copy_2 = customer_completes_journey.copy()
main_process = gen.partial_order(dependencies=[(initial_selection, booking_and_updates), (booking_and_updates, optional_reminders), (optional_reminders, customer_completes_journey_copy_2)])


# Final model including optional post-travel feedback
customer_completes_journey_copy_3 = customer_completes_journey.copy()
final_model = gen.partial_order(dependencies=[(main_process, customer_completes_journey_copy_3), (customer_completes_journey_copy_3, post_travel_feedback_or_services)])
