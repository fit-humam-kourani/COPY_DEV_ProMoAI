from utils.model_generation import ModelGenerator

gen = ModelGenerator()

# Define activities
enjoy = gen.activity('Enjoy Spaghetti Carbonara')
plate = gen.activity('Plate and garnish dish')
mix = gen.activity('Mix with egg mixture and pancetta')
pour = gen.activity('Pour egg yolk and cheese mixture')
boil_spaghetti = gen.activity('Boil spaghetti')
whisk = gen.activity('Whisk egg yolks, Parmesan, and black pepper')
saute = gen.activity('Saute pancetta until crispy')
boil_salted_water = gen.activity('Boil salted water')
gather = gen.activity('Gather ingredients')
plan = gen.activity('Plan how to cook')

# Define dependencies
dependencies = [
    (enjoy, plate),
    (plate, mix),
    (mix, pour),
    (pour, boil_spaghetti),
    (boil_spaghetti, whisk),
    (boil_spaghetti, saute),
    (whisk, boil_salted_water),
    (saute, boil_salted_water),
    (whisk, gather),
    (saute, gather),
    (boil_salted_water, plan),
    (gather, plan),
]

# Create the final partial order model
final_model = gen.partial_order(dependencies=dependencies)
