from utils.model_generation import ModelGenerator
gen = ModelGenerator()

# Activities
plan = gen.activity('Plan how to cook')
boil_salted_water = gen.activity('Boil salted water')
gather_ingredients = gen.activity('Gather ingredients')
whisk_egg_yolks = gen.activity('Whisk egg yolks, Parmesan, and black pepper')
saute_pancetta = gen.activity('Saute pancetta until crispy')
boil_spaghetti = gen.activity('Boil spaghetti')
pour_egg_mixture = gen.activity('Pour egg yolk and cheese mixture')
mix_with_pancetta = gen.activity('Mix with egg mixture and pancetta')
plate_and_garnish = gen.activity('Plate and garnish dish')
enjoy = gen.activity('Enjoy Spaghetti Carbonara')

# Dependencies
dependencies = [
    (plan, boil_salted_water),
    (plan, gather_ingredients),
    (boil_salted_water, boil_spaghetti),
    (gather_ingredients, whisk_egg_yolks),
    (gather_ingredients, saute_pancetta),
    (whisk_egg_yolks, pour_egg_mixture),
    (boil_spaghetti, pour_egg_mixture),
    (pour_egg_mixture, mix_with_pancetta),
    (saute_pancetta, mix_with_pancetta),
    (mix_with_pancetta, plate_and_garnish),
    (plate_and_garnish, enjoy)
]

final_model = gen.partial_order(dependencies=dependencies)
