from utils.model_generation import ModelGenerator

gen = ModelGenerator()

# Define activities
plan = gen.activity('Plan how to cook')
gather = gen.activity('Gather ingredients')
boil_water = gen.activity('Boil salted water')
saute_pancetta = gen.activity('Saute pancetta until crispy')
whisk_eggs = gen.activity('Whisk egg yolks, Parmesan, and black pepper')
boil_spaghetti = gen.activity('Boil spaghetti')
pour_mixture = gen.activity('Pour egg yolk and cheese mixture')
mix = gen.activity('Mix with egg mixture and pancetta')
plate = gen.activity('Plate and garnish dish')
enjoy = gen.activity('Enjoy Spaghetti Carbonara')

# Create the process model
final_model = gen.partial_order(dependencies=[
    (plan, gather),
    (gather, boil_water),
    (gather, saute_pancetta),
    (gather, whisk_eggs),
    (boil_water, boil_spaghetti),
    (boil_spaghetti, pour_mixture),
    (saute_pancetta, mix),
    (whisk_eggs, pour_mixture),
    (pour_mixture, mix),
    (mix, plate),
    (plate, enjoy)
])
